#include <Arduino.h>

#define RELE_PIN 23

void setup() {
  Serial.begin(115200);
  delay(1000);

  pinMode(RELE_PIN, OUTPUT);

  // Módulo Low Level Trigger:
  // HIGH = desligado
  // LOW = ligado
  digitalWrite(RELE_PIN, HIGH);

  Serial.println("Teste do rele iniciado");
}

void loop() {
  Serial.println("Ligando rele...");
  digitalWrite(RELE_PIN, LOW);
  delay(3000);

  Serial.println("Desligando rele...");
  digitalWrite(RELE_PIN, HIGH);
  delay(3000);
}
