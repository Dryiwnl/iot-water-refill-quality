#include <Arduino.h>

#define PH_PIN 34

// Calibrado considerando agua limpa em 1.725 V como pH 7
float calibration_value = 16.83;

// Tempo de cada teste
const unsigned long TEMPO_TESTE = 180000; // 3 minutos

// Intervalo entre as leituras
const unsigned long INTERVALO_LEITURA = 1000; // 1 segundo

int buffer_arr[10], temp;

int lerADCFiltrado() {
  for (int i = 0; i < 10; i++) {
    buffer_arr[i] = analogRead(PH_PIN);
    delay(30);
  }

  // Ordena os valores
  for (int i = 0; i < 9; i++) {
    for (int j = i + 1; j < 10; j++) {
      if (buffer_arr[i] > buffer_arr[j]) {
        temp = buffer_arr[i];
        buffer_arr[i] = buffer_arr[j];
        buffer_arr[j] = temp;
      }
    }
  }

  long soma = 0;

  // Descarta os 2 menores e os 2 maiores
  // Usa os 6 valores centrais
  for (int i = 2; i < 8; i++) {
    soma += buffer_arr[i];
  }

  return soma / 6;
}

float lerTensaoPH() {
  int adc = lerADCFiltrado();

  // Conversao para tensao no ESP32
  float tensao = adc * 3.3 / 4095.0;

  return tensao;
}

float calcularPH(float tensao) {
  float ph = -5.70 * tensao + calibration_value;

  if (ph < 0) {
    ph = 0;
  }

  if (ph > 14) {
    ph = 14;
  }

  return ph;
}

float executarTeste(const char* nomeAmostra, float &mediaTensao) {
  Serial.println();
  Serial.print("=== INICIANDO TESTE: ");
  Serial.print(nomeAmostra);
  Serial.println(" ===");

  unsigned long inicio = millis();
  unsigned long ultimaLeitura = 0;

  float somaPH = 0;
  float somaTensao = 0;
  int quantidadeLeituras = 0;

  while (millis() - inicio < TEMPO_TESTE) {
    if (millis() - ultimaLeitura >= INTERVALO_LEITURA) {
      ultimaLeitura = millis();

      float tensao = lerTensaoPH();
      float ph = calcularPH(tensao);

      somaTensao += tensao;
      somaPH += ph;
      quantidadeLeituras++;

      unsigned long tempoDecorrido = (millis() - inicio) / 1000;
      unsigned long tempoRestante = (TEMPO_TESTE / 1000) - tempoDecorrido;

      Serial.print(nomeAmostra);
      Serial.print(" | Tempo restante: ");
      Serial.print(tempoRestante);
      Serial.print(" s");

      Serial.print(" | Tensao: ");
      Serial.print(tensao, 3);
      Serial.print(" V");

      Serial.print(" | pH: ");
      Serial.println(ph, 2);
    }
  }

  mediaTensao = somaTensao / quantidadeLeituras;
  float mediaPH = somaPH / quantidadeLeituras;

  Serial.println();
  Serial.print("Teste finalizado: ");
  Serial.println(nomeAmostra);

  Serial.print("Media de tensao: ");
  Serial.print(mediaTensao, 3);
  Serial.println(" V");

  Serial.print("Media de pH: ");
  Serial.println(mediaPH, 2);

  return mediaPH;
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  analogReadResolution(12);
  analogSetPinAttenuation(PH_PIN, ADC_11db);

  Serial.println("=== TESTE COMPARATIVO pH ===");
  Serial.println("PH-4502C com ESP32");
  Serial.println();
  Serial.println("Etapa 1: coloque a sonda na agua limpa.");
  Serial.println("O teste da agua vai comecar em 10 segundos.");
  delay(10000);

  float tensaoAgua = 0;
  float tensaoLeite = 0;

  float phAgua = executarTeste("AGUA LIMPA", tensaoAgua);

  Serial.println();
  Serial.println("Agora retire a sonda da agua limpa.");
  Serial.println("Lave rapidamente a sonda com agua limpa.");
  Serial.println("Coloque a sonda no leite.");
  Serial.println("O teste do leite vai comecar em 20 segundos.");
  delay(20000);

  float phLeite = executarTeste("LEITE", tensaoLeite);

  Serial.println();
  Serial.println("======================================");
  Serial.println("RESULTADO FINAL DO TESTE");
  Serial.println("======================================");

  Serial.print("Agua limpa -> Tensao media: ");
  Serial.print(tensaoAgua, 3);
  Serial.print(" V | pH medio estimado: ");
  Serial.println(phAgua, 2);

  Serial.print("Leite      -> Tensao media: ");
  Serial.print(tensaoLeite, 3);
  Serial.print(" V | pH medio estimado: ");
  Serial.println(phLeite, 2);

  Serial.println("======================================");
  Serial.println("Teste concluido.");
}

}
